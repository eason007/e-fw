  247.  /******************************************************************************
  248.   * Defines a filename for the initial value of a variable.
  249.   *
  250.   * It may be passed either a varname and a file name as two strings or
  251.   * a hash of strings with the key being the varname and the value
  252.   * being the file name.
  253.   *
  254.   * The new mappings are stored in the array $this->file.
  255.   * The files are not loaded yet, but only when needed.
  256.   *
  257.   * Returns true on success, false on error.
  258.   *
  259.   * usage: set_file(array $filelist = (string $varname => string $filename))
  260.   * or
  261.   * usage: set_file(string $varname, string $filename)
  262.   *
  263.   * @param     $varname      either a string containing a varname or a hash of varname/file name pairs.
  264.   * @param     $filename     if varname is a string this is the filename otherwise filename is not required
  265.   * @access    public
  266.   * @return    boolean
  267.   */
  268.   function set_file($varname, $filename = "") {
  269.     if (!is_array($varname)) {
  270.       if ($this->debug & 4) {
  271.         echo "<p><b>set_file:</b> (with scalar) varname = $varname, filename = $filename</p>\n";
  272.       }
  273.       if ($filename == "") {
  274.         $this->halt("set_file: For varname $varname filename is empty.");
  275.         return false;
  276.       }
  277.       $this->file[$varname] = $this->filename($filename);
  278.     } else {
  279.       reset($varname);
  280.       while(list($v, $f) = each($varname)) {
  281.         if ($this->debug & 4) {
  282.           echo "<p><b>set_file:</b> (with array) varname = $v, filename = $f</p>\n";
  283.         }
  284.         if ($f == "") {
  285.           $this->halt("set_file: For varname $v filename is empty.");
  286.           return false;
  287.         }
  288.         $this->file[$v] = $this->filename($f);
  289.       }
  290.     }
  291.     return true;
  292.   }

  295.  /******************************************************************************
  296.   * A variable $parent may contain a variable block defined by:
  297.   * &lt;!-- BEGIN $varname --&gt; content &lt;!-- END $varname --&gt;. This function removes
  298.   * that block from $parent and replaces it with a variable reference named $name.
  299.   * The block is inserted into the varkeys and varvals hashes. If $name is
  300.   * omitted, it is assumed to be the same as $varname.
  301.   *
  302.   * Blocks may be nested but care must be taken to extract the blocks in order
  303.   * from the innermost block to the outermost block.
  304.   *
  305.   * Returns true on success, false on error.
  306.   *
  307.   * usage: set_block(string $parent, string $varname, [string $name = ""])
  308.   *
  309.   * @param     $parent       a string containing the name of the parent variable
  310.   * @param     $varname      a string containing the name of the block to be extracted
  311.   * @param     $name         the name of the variable in which to store the block
  312.   * @access    public
  313.   * @return    boolean
  314.   */
  315.   function set_block($parent, $varname, $name = "") {
  316.     if ($this->debug & 4) {
  317.       echo "<p><b>set_block:</b> parent = $parent, varname = $varname, name = $name</p>\n";
  318.     }
  319.     if (!$this->loadfile($parent)) {
  320.       $this->halt("set_block: unable to load $parent.");
  321.       return false;
  322.     }
  323.     if ($name == "") {
  324.       $name = $varname;
  325.     }
  326.  
  327.     $str = $this->get_var($parent);
  328.     $reg = "/[ \t]*<!--\s+BEGIN $varname\s+-->\s*?\n?(\s*.*?\n?)\s*<!--\s+END $varname\s+-->\s*?\n?/sm";
  329.     preg_match_all($reg, $str, $m);
  330.     $str = preg_replace($reg, "{" . "$name}", $str);
  331.     $this->set_var($varname, $m[1][0]);
  332.     $this->set_var($parent, $str);
  333.     return true;
  334.   }

  337.  /******************************************************************************
  338.   * This functions sets the value of a variable.
  339.   *
  340.   * It may be called with either a varname and a value as two strings or an
  341.   * an associative array with the key being the varname and the value being
  342.   * the new variable value.
  343.   *
  344.   * The function inserts the new value of the variable into the $varkeys and
  345.   * $varvals hashes. It is not necessary for a variable to exist in these hashes
  346.   * before calling this function.
  347.   *
  348.   * An optional third parameter allows the value for each varname to be appended
  349.   * to the existing variable instead of replacing it. The default is to replace.
  350.   * This feature was introduced after the 7.2d release.
  351.   *
  352.   *
  353.   * usage: set_var(string $varname, [string $value = ""], [boolean $append = false])
  354.   * or
  355.   * usage: set_var(array $varname = (string $varname => string $value), [mixed $dummy_var], [boolean $append = false])
  356.   *
  357.   * @param     $varname      either a string containing a varname or a hash of varname/value pairs.
  358.   * @param     $value        if $varname is a string this contains the new value for the variable otherwise this parameter is ignored
  359.   * @param     $append       if true, the value is appended to the variable's existing value
  360.   * @access    public
  361.   * @return    void
  362.   */
  363.   function set_var($varname, $value = "", $append = false) {
  364.     if (!is_array($varname)) {
  365.       if (!empty($varname)) {
  366.         if ($this->debug & 1) {
  367.           printf("<b>set_var:</b> (with scalar) <b>%s</b> = '%s'<br>\n", $varname, htmlentities($value));
  368.         }
  369.         $this->varkeys[$varname] = "/".$this->varname($varname)."/";
  370.         if ($append && isset($this->varvals[$varname])) {
  371.           $this->varvals[$varname] .= $value;
  372.         } else {
  373.           $this->varvals[$varname] = $value;
  374.         }
  375.       }
  376.     } else {
  377.       reset($varname);
  378.       while(list($k, $v) = each($varname)) {
  379.         if (!empty($k)) {
  380.           if ($this->debug & 1) {
  381.             printf("<b>set_var:</b> (with array) <b>%s</b> = '%s'<br>\n", $k, htmlentities($v));
  382.           }
  383.           $this->varkeys[$k] = "/".$this->varname($k)."/";
  384.           if ($append && isset($this->varvals[$k])) {
  385.             $this->varvals[$k] .= $v;
  386.           } else {
  387.             $this->varvals[$k] = $v;
  388.           }
  389.         }
  390.       }
  391.     }
  392.   }

  395.  /******************************************************************************
  396.   * This functions clears the value of a variable.
  397.   *
  398.   * It may be called with either a varname as a string or an array with the 
  399.   * values being the varnames to be cleared.
  400.   *
  401.   * The function sets the value of the variable in the $varkeys and $varvals 
  402.   * hashes to "". It is not necessary for a variable to exist in these hashes
  403.   * before calling this function.
  404.   *
  405.   *
  406.   * usage: clear_var(string $varname)
  407.   * or
  408.   * usage: clear_var(array $varname = (string $varname))
  409.   *
  410.   * @param     $varname      either a string containing a varname or an array of varnames.
  411.   * @access    public
  412.   * @return    void
  413.   */
  414.   function clear_var($varname) {
  415.     if (!is_array($varname)) {
  416.       if (!empty($varname)) {
  417.         if ($this->debug & 1) {
  418.           printf("<b>clear_var:</b> (with scalar) <b>%s</b><br>\n", $varname);
  419.         }
  420.         $this->set_var($varname, "");
  421.       }
  422.     } else {
  423.       reset($varname);
  424.       while(list($k, $v) = each($varname)) {
  425.         if (!empty($v)) {
  426.           if ($this->debug & 1) {
  427.             printf("<b>clear_var:</b> (with array) <b>%s</b><br>\n", $v);
  428.           }
  429.           $this->set_var($v, "");
  430.         }
  431.       }
  432.     }
  433.   }
  434.  
  435.  
  436.  /******************************************************************************
  437.   * This functions unsets a variable completely.
  438.   *
  439.   * It may be called with either a varname as a string or an array with the 
  440.   * values being the varnames to be cleared.
  441.   *
  442.   * The function removes the variable from the $varkeys and $varvals hashes.
  443.   * It is not necessary for a variable to exist in these hashes before calling
  444.   * this function.
  445.   *
  446.   *
  447.   * usage: unset_var(string $varname)
  448.   * or
  449.   * usage: unset_var(array $varname = (string $varname))
  450.   *
  451.   * @param     $varname      either a string containing a varname or an array of varnames.
  452.   * @access    public
  453.   * @return    void
  454.   */
  455.   function unset_var($varname) {
  456.     if (!is_array($varname)) {
  457.       if (!empty($varname)) {
  458.         if ($this->debug & 1) {
  459.           printf("<b>unset_var:</b> (with scalar) <b>%s</b><br>\n", $varname);
  460.         }
  461.         unset($this->varkeys[$varname]);
  462.         unset($this->varvals[$varname]);
  463.       }
  464.     } else {
  465.       reset($varname);
  466.       while(list($k, $v) = each($varname)) {
  467.         if (!empty($v)) {
  468.           if ($this->debug & 1) {
  469.             printf("<b>unset_var:</b> (with array) <b>%s</b><br>\n", $v);
  470.           }
  471.           unset($this->varkeys[$v]);
  472.           unset($this->varvals[$v]);
  473.         }
  474.       }
  475.     }
  476.   }
  477.  
  478.  
  479.  /******************************************************************************
  480.   * This function fills in all the variables contained within the variable named
  481.   * $varname. The resulting value is returned as the function result and the
  482.   * original value of the variable varname is not changed. The resulting string
  483.   * is not "finished", that is, the unresolved variable name policy has not been
  484.   * applied yet.
  485.   *
  486.   * Returns: the value of the variable $varname with all variables substituted.
  487.   *
  488.   * usage: subst(string $varname)
  489.   *
  490.   * @param     $varname      the name of the variable within which variables are to be substituted
  491.   * @access    public
  492.   * @return    string
  493.   */
  494.   function subst($varname) {
  495.     $varvals_quoted = array();
  496.     if ($this->debug & 4) {
  497.       echo "<p><b>subst:</b> varname = $varname</p>\n";
  498.     }
  499.     if (!$this->loadfile($varname)) {
  500.       $this->halt("subst: unable to load $varname.");
  501.       return false;
  502.     }
  503.  
  504.     // quote the replacement strings to prevent bogus stripping of special chars
  505.     reset($this->varvals);
  506.     while(list($k, $v) = each($this->varvals)) {
  507.       $varvals_quoted[$k] = preg_replace(array('/\\\\/', '/\$/'), array('\\\\\\\\', '\\\\$'), $v);
  508.     }
  509.  
  510.     $str = $this->get_var($varname);
  511.     $str = preg_replace($this->varkeys, $varvals_quoted, $str);
  512.     return $str;
  513.   }
  514.  
  515.  
  516.  /******************************************************************************
  517.   * This is shorthand for print $this->subst($varname). See subst for further
  518.   * details.
  519.   *
  520.   * Returns: always returns false.
  521.   *
  522.   * usage: psubst(string $varname)
  523.   *
  524.   * @param     $varname      the name of the variable within which variables are to be substituted
  525.   * @access    public
  526.   * @return    false
  527.   * @see       subst
  528.   */
  529.   function psubst($varname) {
  530.     if ($this->debug & 4) {
  531.       echo "<p><b>psubst:</b> varname = $varname</p>\n";
  532.     }
  533.     print $this->subst($varname);
  534.  
  535.     return false;
  536.   }
  537.  
  538.  
  539.  /******************************************************************************
  540.   * The function substitutes the values of all defined variables in the variable
  541.   * named $varname and stores or appends the result in the variable named $target.
  542.   *
  543.   * It may be called with either a target and a varname as two strings or a
  544.   * target as a string and an array of variable names in varname.
  545.   *
  546.   * The function inserts the new value of the variable into the $varkeys and
  547.   * $varvals hashes. It is not necessary for a variable to exist in these hashes
  548.   * before calling this function.
  549.   *
  550.   * An optional third parameter allows the value for each varname to be appended
  551.   * to the existing target variable instead of replacing it. The default is to
  552.   * replace.
  553.   *
  554.   * If $target and $varname are both strings, the substituted value of the
  555.   * variable $varname is inserted into or appended to $target.
  556.   *
  557.   * If $handle is an array of variable names the variables named by $handle are
  558.   * sequentially substituted and the result of each substitution step is
  559.   * inserted into or appended to in $target. The resulting substitution is
  560.   * available in the variable named by $target, as is each intermediate step
  561.   * for the next $varname in sequence. Note that while it is possible, it
  562.   * is only rarely desirable to call this function with an array of varnames
  563.   * and with $append = true. This append feature was introduced after the 7.2d
  564.   * release.
  565.   *
  566.   * Returns: the last value assigned to $target.
  567.   *
  568.   * usage: parse(string $target, string $varname, [boolean $append])
  569.   * or
  570.   * usage: parse(string $target, array $varname = (string $varname), [boolean $append])
  571.   *
  572.   * @param     $target      a string containing the name of the variable into which substituted $varnames are to be stored
  573.   * @param     $varname     if a string, the name the name of the variable to substitute or if an array a list of variables to be substituted
  574.   * @param     $append      if true, the substituted variables are appended to $target otherwise the existing value of $target is replaced
  575.   * @access    public
  576.   * @return    string
  577.   * @see       subst
  578.   */
  579.   function parse($target, $varname, $append = false) {
  580.     if (!is_array($varname)) {
  581.       if ($this->debug & 4) {
  582.         echo "<p><b>parse:</b> (with scalar) target = $target, varname = $varname, append = $append</p>\n";
  583.       }
  584.       $str = $this->subst($varname);
  585.       if ($append) {
  586.         $this->set_var($target, $this->get_var($target) . $str);
  587.       } else {
  588.         $this->set_var($target, $str);
  589.       }
  590.     } else {
  591.       reset($varname);
  592.       while(list($i, $v) = each($varname)) {
  593.         if ($this->debug & 4) {
  594.           echo "<p><b>parse:</b> (with array) target = $target, i = $i, varname = $v, append = $append</p>\n";
  595.         }
  596.         $str = $this->subst($v);
  597.         if ($append) {
  598.           $this->set_var($target, $this->get_var($target) . $str);
  599.         } else {
  600.           $this->set_var($target, $str);
  601.         }
  602.       }
  603.     }
  604.  
  605.     if ($this->debug & 4) {
  606.       echo "<p><b>parse:</b> completed</p>\n";
  607.     }
  608.     return $str;
  609.   }
  610.  
  611.  
  612.  /******************************************************************************
  613.   * This is shorthand for print $this->parse(...) and is functionally identical.
  614.   * See parse for further details.
  615.   *
  616.   * Returns: always returns false.
  617.   *
  618.   * usage: pparse(string $target, string $varname, [boolean $append])
  619.   * or
  620.   * usage: pparse(string $target, array $varname = (string $varname), [boolean $append])
  621.   *
  622.   * @param     $target      a string containing the name of the variable into which substituted $varnames are to be stored
  623.   * @param     $varname     if a string, the name the name of the variable to substitute or if an array a list of variables to be substituted
  624.   * @param     $append      if true, the substituted variables are appended to $target otherwise the existing value of $target is replaced
  625.   * @access    public
  626.   * @return    false
  627.   * @see       parse
  628.   */
  629.   function pparse($target, $varname, $append = false) {
  630.     if ($this->debug & 4) {
  631.       echo "<p><b>pparse:</b> passing parameters to parse...</p>\n";
  632.     }
  633.     print $this->finish($this->parse($target, $varname, $append));
  634.     return false;
  635.   }

  637.  
  638.  /******************************************************************************
  639.   * This function returns an associative array of all defined variables with the
  640.   * name as the key and the value of the variable as the value.
  641.   *
  642.   * This is mostly useful for debugging. Also note that $this->debug can be used
  643.   * to echo all variable assignments as they occur and to trace execution.
  644.   *
  645.   * Returns: a hash of all defined variable values keyed by their names.
  646.   *
  647.   * usage: get_vars()
  648.   *
  649.   * @access    public
  650.   * @return    array
  651.   * @see       $debug
  652.   */
  653.   function get_vars() {
  654.     if ($this->debug & 4) {
  655.       echo "<p><b>get_vars:</b> constructing array of vars...</p>\n";
  656.     }
  657.     reset($this->varkeys);
  658.     while(list($k, $v) = each($this->varkeys)) {
  659.       $result[$k] = $this->get_var($k);
  660.     }
  661.     return $result;
  662.   }
  663.  
  664.  
  665.  /******************************************************************************
  666.   * This function returns the value of the variable named by $varname.
  667.   * If $varname references a file and that file has not been loaded yet, the
  668.   * variable will be reported as empty.
  669.   *
  670.   * When called with an array of variable names this function will return a a
  671.   * hash of variable values keyed by their names.
  672.   *
  673.   * Returns: a string or an array containing the value of $varname.
  674.   *
  675.   * usage: get_var(string $varname)
  676.   * or
  677.   * usage: get_var(array $varname)
  678.   *
  679.   * @param     $varname     if a string, the name the name of the variable to get the value of, or if an array a list of variables to return the value of
  680.   * @access    public
  681.   * @return    string or array
  682.   */
  683.   function get_var($varname) {
  684.     if (!is_array($varname)) {
  685.       if (isset($this->varvals[$varname])) {
  686.         $str = $this->varvals[$varname];
  687.       } else {
  688.         $str = "";
  689.       }
  690.       if ($this->debug & 2) {
  691.         printf ("<b>get_var</b> (with scalar) <b>%s</b> = '%s'<br>\n", $varname, htmlentities($str));
  692.       }
  693.       return $str;
  694.     } else {
  695.       reset($varname);
  696.       while(list($k, $v) = each($varname)) {
  697.         if (isset($this->varvals[$v])) {
  698.           $str = $this->varvals[$v];
  699.         } else {
  700.           $str = "";
  701.         }
  702.         if ($this->debug & 2) {
  703.           printf ("<b>get_var:</b> (with array) <b>%s</b> = '%s'<br>\n", $v, htmlentities($str));
  704.         }
  705.         $result[$v] = $str;
  706.       }
  707.       return $result;
  708.     }
  709.   }
  710.  
  711.  
  712.  /******************************************************************************
  713.   * This function returns a hash of unresolved variable names in $varname, keyed
  714.   * by their names (that is, the hash has the form $a[$name] = $name).
  715.   *
  716.   * Returns: a hash of varname/varname pairs or false on error.
  717.   *
  718.   * usage: get_undefined(string $varname)
  719.   *
  720.   * @param     $varname     a string containing the name the name of the variable to scan for unresolved variables
  721.   * @access    public
  722.   * @return    array
  723.   */
  724.   function get_undefined($varname) {
  725.     if ($this->debug & 4) {
  726.       echo "<p><b>get_undefined:</b> varname = $varname</p>\n";
  727.     }
  728.     if (!$this->loadfile($varname)) {
  729.       $this->halt("get_undefined: unable to load $varname.");
  730.       return false;
  731.     }
  732.  
  733.     preg_match_all("/{([^ \t\r\n}]+)}/", $this->get_var($varname), $m);
  734.     $m = $m[1];
  735.     if (!is_array($m)) {
  736.       return false;
  737.     }
  738.  
  739.     reset($m);
  740.     while(list($k, $v) = each($m)) {
  741.       if (!isset($this->varkeys[$v])) {
  742.         if ($this->debug & 4) {
  743.          echo "<p><b>get_undefined:</b> undefined: $v</p>\n";
  744.         }
  745.         $result[$v] = $v;
  746.       }
  747.     }
  748.  
  749.     if (count($result)) {
  750.       return $result;
  751.     } else {
  752.       return false;
  753.     }
  754.   }
  755.  
  756.  
  757.  /******************************************************************************
  758.   * This function returns the finished version of $str. That is, the policy
  759.   * regarding unresolved variable names will be applied to $str.
  760.   *
  761.   * Returns: a finished string derived from $str and $this->unknowns.
  762.   *
  763.   * usage: finish(string $str)
  764.   *
  765.   * @param     $str         a string to which to apply the unresolved variable policy
  766.   * @access    public
  767.   * @return    string
  768.   * @see       set_unknowns
  769.   */
  770.   function finish($str) {
  771.     switch ($this->unknowns) {
  772.       case "keep":
  773.       break;
  774.  
  775.       case "remove":
  776.         $str = preg_replace('/{[^ \t\r\n}]+}/', "", $str);
  777.       break;
  778.  
  779.       case "comment":
  780.         $str = preg_replace('/{([^ \t\r\n}]+)}/', "<!-- Template variable \\1 undefined -->", $str);
  781.       break;
  782.     }
  783.  
  784.     return $str;
  785.   }
  786.  
  787.  
  788.  /******************************************************************************
  789.   * This function prints the finished version of the value of the variable named
  790.   * by $varname. That is, the policy regarding unresolved variable names will be
  791.   * applied to the variable $varname then it will be printed.
  792.   *
  793.   * usage: p(string $varname)
  794.   *
  795.   * @param     $varname     a string containing the name of the variable to finish and print
  796.   * @access    public
  797.   * @return    void
  798.   * @see       set_unknowns
  799.   * @see       finish
  800.   */
  801.   function p($varname) {
  802.     print $this->finish($this->get_var($varname));
  803.   }
  804.  
  805.  
  806.  /******************************************************************************
  807.   * This function returns the finished version of the value of the variable named
  808.   * by $varname. That is, the policy regarding unresolved variable names will be
  809.   * applied to the variable $varname and the result returned.
  810.   *
  811.   * Returns: a finished string derived from the variable $varname.
  812.   *
  813.   * usage: get(string $varname)
  814.   *
  815.   * @param     $varname     a string containing the name of the variable to finish
  816.   * @access    public
  817.   * @return    void
  818.   * @see       set_unknowns
  819.   * @see       finish
  820.   */
  821.   function get($varname) {

  871.  /******************************************************************************
  872.   * If a variable's value is undefined and the variable has a filename stored in
  873.   * $this->file[$varname] then the backing file will be loaded and the file's
  874.   * contents will be assigned as the variable's value.
  875.   *
  876.   * Note that the behaviour of this function changed slightly after the 7.2d
  877.   * release. Where previously a variable was reloaded from file if the value
  878.   * was empty, now this is not done. This allows a variable to be loaded then
  879.   * set to "", and also prevents attempts to load empty variables. Files are
  880.   * now only loaded if $this->varvals[$varname] is unset.
  881.   *
  882.   * Returns: true on success, false on error.
  883.   *
  884.   * usage: loadfile(string $varname)
  885.   *
  886.   * @param     $varname    a string containing the name of a variable to load
  887.   * @access    private
  888.   * @return    boolean
  889.   * @see       set_file
  890.   */
  891.   function loadfile($varname) {
  892.     if ($this->debug & 4) {
  893.       echo "<p><b>loadfile:</b> varname = $varname</p>\n";
  894.     }
  895.  
  896.     if (!isset($this->file[$varname])) {
  897.       // $varname does not reference a file so return
  898.       if ($this->debug & 4) {
  899.         echo "<p><b>loadfile:</b> varname $varname does not reference a file</p>\n";
  900.       }
  901.       return true;
  902.     }
  903.  
  904.     if (isset($this->varvals[$varname])) {
  905.       // will only be unset if varname was created with set_file and has never been loaded
  906.       // $varname has already been loaded so return
  907.       if ($this->debug & 4) {
  908.         echo "<p><b>loadfile:</b> varname $varname is already loaded</p>\n";
  909.       }
  910.       return true;
  911.     }
  912.     $filename = $this->file[$varname];
  913.  
  914.     /* use @file here to avoid leaking filesystem information if there is an error */
  915.     $str = implode("", @file($filename));
  916.     if (empty($str)) {
  917.       $this->halt("loadfile: While loading $varname, $filename does not exist or is empty.");
  918.       return false;
  919.     }
  920.  
  921.     if ($this->debug & 4) {
  922.       printf("<b>loadfile:</b> loaded $filename into $varname<br>\n");
  923.     }
  924.     $this->set_var($varname, $str);
  925.  
  926.     return true;
  927.   }

  931.   * This function is called whenever an error occurs and will handle the error

  935.   * Returns: always returns false.

  937.   * usage: halt(string $msg)

  940.   * @access    private
  941.   * @return    void
  942.   * @see       $halt_on_error
  943.   */
  944.   function halt($msg) {
  945.     $this->last_error = $msg;
  946.  
  947.     if ($this->halt_on_error != "no") {
  948.       $this->haltmsg($msg);

  952.       die("<b>Halted.</b>");

  955.     return false;
  956.   }
