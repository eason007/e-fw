  203.     * It may be called with either a varname and a value as two strings or an

  217.     * or

  223.     * @param     $append       if true, the value is appended to the variable's existing value

  227.     function setVar($varname, $value = "", $append = false) {
  228.         if (!is_array($varname)) {
  229.             if (!empty($varname)) {
  230.                 if ($this->debug & 1) {
  231.                     printf("<b>setVar:</b> (with scalar) <b>%s</b> = '%s'<br>\n", $varname, htmlentities($value));

  233.                 $this->varkeys[$varname] = "/".$this->varname($varname)."/";
  234.                 if ($append && isset($this->varvals[$varname])) {
  235.                     $this->varvals[$varname] .= $value;
  236.                 } else {
  237.                     $this->varvals[$varname] = $value;
  238.                 }
  239.             }
  240.         } else {
  241.             reset($varname);
  242.             while(list($k, $v) = each($varname)) {
  243.                 if (!empty($k)) {
  244.                     if ($this->debug & 1) {
  245.                         printf("<b>setVar:</b> (with array) <b>%s</b> = '%s'<br>\n", $k, htmlentities($v));
  246.                     }
  247.                     $this->varkeys[$k] = "/".$this->varname($k)."/";
  248.                     if ($append && isset($this->varvals[$k])) {
  249.                         $this->varvals[$k] .= $v;
  250.                     } else {
  251.                         $this->varvals[$k] = $v;
  252.                     }
  253.                 }
  254.             }
  255.         }

  364.     function subst($varname) {
  365.         $varvals_quoted = array();
  366.         if ($this->debug & 4) {
  367.             echo "<p><b>subst:</b> varname = $varname</p>\n";
  368.         }
  369.         if (!$this->loadfile($varname)) {
  370.             $this->halt("subst: unable to load $varname.");
  371.             return false;
  372.         }
  373.  
  374.         // quote the replacement strings to prevent bogus stripping of special chars
  375.         reset($this->varvals);
  376.         while(list($k, $v) = each($this->varvals)) {
  377.             $varvals_quoted[$k] = preg_replace(array('/\\\\/', '/\$/'), array('\\\\\\\\', '\\\\$'), $v);
  378.         }
  379.  
  380.         $str = $this->getVar($varname);
  381.         $str = preg_replace($this->varkeys, $varvals_quoted, $str);
  382.         return $str;

  412.      * <pre>
  413.     * The function substitutes the values of all defined variables in the variable
  414.     * named $varname and stores or appends the result in the variable named $target.
  415.     *
  416.     * It may be called with either a target and a varname as two strings or a
  417.     * target as a string and an array of variable names in varname.
  418.     *
  419.     * The function inserts the new value of the variable into the $varkeys and
  420.     * $varvals hashes. It is not necessary for a variable to exist in these hashes
  421.     * before calling this function.
  422.     *
  423.     * An optional third parameter allows the value for each varname to be appended
  424.     * to the existing target variable instead of replacing it. The default is to
  425.     * replace.
  426.     *
  427.     * If $target and $varname are both strings, the substituted value of the
  428.     * variable $varname is inserted into or appended to $target.
  429.     *
  430.     * If $handle is an array of variable names the variables named by $handle are
  431.     * sequentially substituted and the result of each substitution step is
  432.     * inserted into or appended to in $target. The resulting substitution is
  433.     * available in the variable named by $target, as is each intermediate step
  434.     * for the next $varname in sequence. Note that while it is possible, it
  435.     * is only rarely desirable to call this function with an array of varnames
  436.     * and with $append = true. This append feature was introduced after the 7.2d
  437.     * release.
  438.     *
  439.     * Returns: the last value assigned to $target.
  440.     *
  441.     * usage: parse(string $target, string $varname, [boolean $append])
  442.     * or
  443.     * usage: parse(string $target, array $varname = (string $varname), [boolean $append])
  444.     * </pre>
  445.     *
  446.     * @param     $target      a string containing the name of the variable into which substituted $varnames are to be stored
  447.     * @param     $varname     if a string, the name the name of the variable to substitute or if an array a list of variables to be substituted
  448.     * @param     $append      if true, the substituted variables are appended to $target otherwise the existing value of $target is replaced
  449.     * @access    public
  450.     * @return    string 
  451.     * @see       subst
  452.     */
  453.     function parse($target, $varname, $append = false) {
  454.         if (!is_array($varname)) {
  455.             if ($this->debug & 4) {
  456.                 echo "<p><b>parse:</b> (with scalar) target = $target, varname = $varname, append = $append</p>\n";
  457.             }
  458.             $str = $this->subst($varname);
  459.             if ($append) {
  460.                 $this->setVar($target, $this->getVar($target) . $str);
  461.             } else {
  462.                 $this->setVar($target, $str);
  463.             }
  464.         } else {
  465.             reset($varname);
  466.             while(list($i, $v) = each($varname)) {
  467.                 if ($this->debug & 4) {
  468.                     echo "<p><b>parse:</b> (with array) target = $target, i = $i, varname = $v, append = $append</p>\n";
  469.                 }
  470.                 $str = $this->subst($v);
  471.                 if ($append) {
  472.                     $this->setVar($target, $this->getVar($target) . $str);
  473.                 } else {
  474.                     $this->setVar($target, $str);
  475.                 }
  476.             }
  477.         }
  478.  
  479.         if ($this->debug & 4) {
  480.             echo "<p><b>parse:</b> completed</p>\n";
  481.         }
  482.         return $str;
  483.  
  484.     }
  485.  
  486.  
  487.     /**
  488.      * <pre>
  489.     * This is shorthand for print $this->parse(...) and is functionally identical.
  490.     * See parse for further details.
  491.     *
  492.     * Returns: always returns false.
  493.     *
  494.     * usage: pparse(string $target, string $varname, [boolean $append])
  495.     * or
  496.     * usage: pparse(string $target, array $varname = (string $varname), [boolean $append])
  497.     * </pre>
  498.     *
  499.     * @param     $target      a string containing the name of the variable into which substituted $varnames are to be stored
  500.     * @param     $varname     if a string, the name the name of the variable to substitute or if an array a list of variables to be substituted
  501.     * @param     $append      if true, the substituted variables are appended to $target otherwise the existing value of $target is replaced
  502.     * @access    public
  503.     * @return    false 
  504.     * @see       parse
  505.     */
  506.     function pparse($target, $varname, $append = false) {
  507.         if ($this->debug & 4) {
  508.             echo "<p><b>pparse:</b> passing parameters to parse...</p>\n";
  509.         }
  510.         print $this->finish($this->parse($target, $varname, $append));
  511.         return false;
  512.     }

  514.  
  515.     /**
  516.      * <pre>
  517.     * This function returns an associative array of all defined variables with the
  518.     * name as the key and the value of the variable as the value.
  519.     *
  520.     * This is mostly useful for debugging. Also note that $this->debug can be used
  521.     * to echo all variable assignments as they occur and to trace execution.
  522.     *
  523.     * Returns: a hash of all defined variable values keyed by their names.
  524.     *
  525.     * usage: getVars()
  526.     * </pre>
  527.     *
  528.     * @access    public
  529.     * @return    array 
  530.     * @see       $debug
  531.     */
  532.     function getVars() {
  533.         if ($this->debug & 4) {
  534.             echo "<p><b>getVars:</b> constructing array of vars...</p>\n";
  535.         }
  536.         reset($this->varkeys);
  537.         while(list($k, $v) = each($this->varkeys)) {
  538.             $result[$k] = $this->getVar($k);
  539.         }
  540.         return $result;
  541.     }
  542.  
  543.  
  544.     /**
  545.      * <pre>
  546.     * This function returns the value of the variable named by $varname.
  547.     * If $varname references a file and that file has not been loaded yet, the
  548.     * variable will be reported as empty.
  549.     *
  550.     * When called with an array of variable names this function will return a a
  551.     * hash of variable values keyed by their names.
  552.     *
  553.     * Returns: a string or an array containing the value of $varname.
  554.     *
  555.     * usage: getVar(string $varname)
  556.     * or
  557.     * usage: getVar(array $varname)
  558.     * </pre>
  559.     *
  560.     * @param     $varname     if a string, the name the name of the variable to get the value of, or if an array a list of variables to return the value of
  561.     * @access    public
  562.     * @return    string or array
  563.     */
  564.     function getVar($varname) {
  565.         if (!is_array($varname)) {
  566.             if (isset($this->varvals[$varname])) {
  567.                 $str = $this->varvals[$varname];
  568.             } else {
  569.                 $str = "";
  570.             }
  571.             if ($this->debug & 2) {
  572.                 printf ("<b>getVar</b> (with scalar) <b>%s</b> = '%s'<br>\n", $varname, htmlentities($str));
  573.             }
  574.             return $str;
  575.         } else {
  576.             reset($varname);
  577.             while(list($k, $v) = each($varname)) {
  578.                 if (isset($this->varvals[$v])) {
  579.                     $str = $this->varvals[$v];
  580.                 } else {
  581.                     $str = "";
  582.                 }
  583.                 if ($this->debug & 2) {
  584.                     printf ("<b>getVar:</b> (with array) <b>%s</b> = '%s'<br>\n", $v, htmlentities($str));
  585.                 }
  586.                 $result[$v] = $str;
  587.             }
  588.             return $result;
  589.         }
  590.     }

  592.  
  593.     /**
  594.      * <pre>
  595.     * This function returns a hash of unresolved variable names in $varname, keyed
  596.     * by their names (that is, the hash has the form $a[$name] = $name).
  597.     *
  598.     * Returns: a hash of varname/varname pairs or false on error.
  599.     *
  600.     * usage: getUndefined(string $varname, string $keyword)
  601.     * </pre>
  602.     *
  603.     * @param     $varname     a string containing the name the name of the variable to scan for unresolved variables
  604.     * @param      $varname 
  605.     * @access    public
  606.     * @return    array 
  607.     */
  608.     function getUndefined($varname, $keyword = '') {
  609.         if ($this->debug & 4) {
  610.             echo "<p><b>getUndefined:</b> varname = $varname</p>\n";
  611.         }
  612.         if(!$this->getVar($varname)) {
  613.             if (!$this->loadfile($varname)) {
  614.                 $this->halt("getUndefined: unable to load $varname.");
  615.                 return false;
  616.             }
  617.         }
  618.         $keyword = str_replace("(", "\(",str_replace(")", "\)", $keyword));
  619.         preg_match_all("/{((".$keyword.")[^ \t\r\n}]+)}/", $this->getVar($varname), $m);
  620.         //print_r($m);
  621.         $m = $m[1];
  622.         if (!is_array($m)) {
  623.             return false;
  624.         }
  625.  
  626.         reset($m);
  627.         while(list($k, $v) = each($m)) {
  628.             if (!isset($this->varkeys[$v])) {
  629.                 if ($this->debug & 4) {
  630.                     echo "<p><b>getUndefined:</b> undefined: $v</p>\n";
  631.                 }
  632.                 $result[$v] = $v;
  633.             }
  634.         }
  635.  
  636.         if (count($result)) {
  637.             return $result;
  638.         } else {
  639.             return false;
  640.         }
  641.     }
  642.  
  643.  
  644.     /**
  645.      * <pre>
  646.     * This function returns the finished version of $str. That is, the policy
  647.     * regarding unresolved variable names will be applied to $str.
  648.     *
  649.     * Returns: a finished string derived from $str and $this->unknowns.
  650.     *
  651.     * usage: finish(string $str)
  652.     * </pre>
  653.     *
  654.     * @param     $str         a string to which to apply the unresolved variable policy
  655.     * @access    public
  656.     * @return    string 
  657.     * @see       setUnknowns
  658.     */
  659.     function finish($str) {
  660.         switch ($this->unknowns) {
  661.             case "keep":
  662.             break;
  663.  
  664.             case "remove":
  665.             $str = preg_replace('/{[^ \t\r\n}]+}/', "", $str);
  666.             break;
  667.  
  668.             case "comment":
  669.             $str = preg_replace('/{([^ \t\r\n}]+)}/', "<!-- Template variable \\1 undefined -->", $str);
  670.             break;
  671.         }
  672.  
  673.         return $str;
  674.     }
  675.  
  676.  
  677.     /**
  678.      * <pre>
  679.     * This function prints the finished version of the value of the variable named
  680.     * by $varname. That is, the policy regarding unresolved variable names will be
  681.     * applied to the variable $varname then it will be printed.
  682.     *
  683.     * usage: p(string $varname , blooean $gz )
  684.     * </pre>
  685.     *
  686.     * @param     $varname     a string containing the name of the variable to finish and print
  687.     * @param     $gz 
  688.     * @access    public
  689.     * @return    void 
  690.     * @see       setUnknowns
  691.     * @see       finish
  692.     */
  693.     function p($varname , $gz = false) {
  694.         if($gz) {
  695.             ob_start();
  696.             ob_implicit_flush(0);
  697.             echo str_replace(urlencode("}"),"}",str_replace(urlencode("{"),"{",$this->finish($this->getVar($varname))));
  698.             GZIP_CLASS::gzDocOut();
  699.             unset($varname);            
  700.             exit();
  701.         } else {
  702.             echo str_replace(urlencode("}"),"}",str_replace(urlencode("{"),"{",$this->finish($this->getVar($varname))));
  703.             unset($varname);
  704.             exit();
  705.         }        
  706.     }

  709.     /**
  710.      * <pre>
  711.     * This function returns the finished version of the value of the variable named
  712.     * by $varname. That is, the policy regarding unresolved variable names will be
  713.     * applied to the variable $varname and the result returned.
  714.     *
  715.     * Returns: a finished string derived from the variable $varname.
  716.     *
  717.     * usage: get(string $varname)
  718.     * </pre>
  719.     *
  720.     * @param     $varname     a string containing the name of the variable to finish
  721.     * @access    public

  724.     * @see       finish
  725.     */
  726.     function get($varname) {

  728.     }

  730.  
  731.     /**
  732.      * <pre>
  733.     * When called with a relative pathname, this function will return the pathname
  734.     * with $this->root prepended. Absolute pathnames are returned unchanged.
  735.     *
  736.     * Returns: a string containing an absolute pathname.
  737.     *
  738.     * usage: filename(string $filename)
  739.     * </pre>
  740.     *
  741.     * @param     $filename    a string containing a filename
  742.     * @access    private
  743.     * @return    string 
  744.     * @see       setRoot
  745.     */
  746.     function filename($filename) {
  747.         if ($this->debug & 4) {

  749.         }
  750.         if (substr($filename, 0, 1) != "/") {
  751.             $filename = $this->root."/".$filename;
  752.         }
  753.         if (!file_exists($filename)) {
  754.             $this->halt("filename: file $filename does not exist.");

  756.  
  757.         return $filename;
  758.     }
  759.  
  760.  
  761.     /**
  762.      * <pre>
  763.     * This function will construct a regexp for a given variable name with any
  764.     * special chars quoted.
  765.     *
  766.     * Returns: a string containing an escaped variable name.
  767.     *
  768.     * usage: varname(string $varname)
  769.     * </pre>
  770.     *
  771.     * @param     $varname    a string containing a variable name
  772.     * @access    private
  773.     * @return    string 
  774.     */
  775.     function varname($varname) {
  776.         return preg_quote("{".$varname."}");
  777.     }
  778.  
  779.  
  780.     /**
  781.      * <pre>
  782.     * If a variable's value is undefined and the variable has a filename stored in
  783.     * $this->file[$varname] then the backing file will be loaded and the file's
  784.     * contents will be assigned as the variable's value.
  785.     *
  786.     * Note that the behaviour of this function changed slightly after the 7.2d
  787.     * release. Where previously a variable was reloaded from file if the value
  788.     * was empty, now this is not done. This allows a variable to be loaded then
  789.     * set to "", and also prevents attempts to load empty variables. Files are
  790.     * now only loaded if $this->varvals[$varname] is unset.
  791.     *
  792.     * Returns: true on success, false on error.
  793.     *
  794.     * usage: loadfile(string $varname)
  795.     * </pre>
  796.     *
  797.     * @param     $varname    a string containing the name of a variable to load
  798.     * @access    private
  799.     * @return    boolean 
  800.     * @see       setFile
  801.     */
  802.     function loadfile($varname) {
  803.         if ($this->debug & 4) {
  804.             echo "<p><b>loadfile:</b> varname = $varname</p>\n";
  805.         }
  806.         if (!isset($this->file[$varname])) {
  807.             // $varname does not reference a file so return
  808.             if ($this->debug & 4) {
  809.                 echo "<p><b>loadfile:</b> varname $varname does not reference a file</p>\n";
  810.             }
  811.             return true;
  812.         }
  813.  
  814.         if (isset($this->varvals[$varname])) {
  815.             // will only be unset if varname was created with setFile and has never been loaded
  816.             // $varname has already been loaded so return
  817.             if ($this->debug & 4) {
  818.                 echo "<p><b>loadfile:</b> varname $varname is already loaded</p>\n";
  819.             }
