  154.  
  155.         $controller = E_FW::load_Class($controllerName);
  156.         if (!$controller) {
  157.             return false;
  158.         }
  159.  
  160.         if (!method_exists($controller, $actionMethod)) {
  161.             return false;
  162.         }
  163.         else{
  164.             if (method_exists($controller, '_beforeExecute')) {
  165.                 $controller->_beforeExecute($actionMethod);
  166.             }
  167.  
  168.             $ret = $controller->{$actionMethod}($loadParam);
  169.  
  170.             if (method_exists($controller, '_afterExecute')) {
  171.                 $controller->_afterExecute($actionMethod);
  172.             }
  173.  
  174.             return $ret;
  175.         }

  309.                 
  310.                 if ( (is_array($val)) and (is_array($tmp)) ){

  397.     private function get_FilePath($filename)
  398.     {
  399.         if (E_FW::get_Config('SEARCH_FILE_NAME/'.$filename)) {
