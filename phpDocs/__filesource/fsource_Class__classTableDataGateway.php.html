  400.  
  401.         $this->clear();

  403.         if (!$params['isExecute']) {

  407.         if (!$this->_beforeInsert($rowData)) {

  421.             
  422.  
  423.             if (!empty($linkData)){
  424.                 $isFound = false;
  425.                 
  426.                 foreach($linkData as $key => $val){
  427.                     $result[$key] = $this->_insertLinkData($key, $val, $result[$this->primaryKey]);
  428.  
  429.                     if ($result[$key] == 0){

  431.  
  432.                         $result[$this->primaryKey] = 0;
  433.                         
  434.                         $isFound = true;
  435.  
  436.                         break;
  437.                     }
  438.                 }
  439.  
  440.                 if (!$isFound){
  441.                     $this->db->commitT();
  442.                 }

  446.         return $result;
  447.     }
